// ===== SETTINGS =====
const WHATSAPP_NUMBER = "923064572592"; // digits only
const GOOGLE_MAP_LINK = "https://maps.app.goo.gl/tDP4XwHJ9YUrTsdN8";

// ===== Helpers =====
function openWhatsApp(message) {
  const encoded = encodeURIComponent(message);
  const url = `https://wa.me/${WHATSAPP_NUMBER}?text=${encoded}`;
  window.open(url, "_blank", "noopener");
}

// ===== Init =====
const yearEl = document.getElementById("year");
if (yearEl) yearEl.textContent = new Date().getFullYear();

// Map link
const mapLinkEl = document.getElementById("mapLink");
if (mapLinkEl) mapLinkEl.href = GOOGLE_MAP_LINK;

// Default WhatsApp message
const defaultChatText =
  `Assalam o Alaikum Doctor Sahib,\n` +
  `Mujhe appointment/consultation ke liye guidance chahiye.\n\n` +
  `Clinic: Lahore Care Hospital\n` +
  `Timing: Mon–Sat 04:00 PM – 07:00 PM\n\n` +
  `JazakAllah`;

// Attach WhatsApp openers if present
["waFloat","openWhatsAppTop","openWhatsAppMid","openWhatsAppBottom","openWhatsAppContact"].forEach(id=>{
  const el = document.getElementById(id);
  if (!el) return;
  el.addEventListener("click", (e) => {
    e.preventDefault();
    openWhatsApp(defaultChatText);
  });
});

// Appointment form -> WhatsApp
const form = document.getElementById("appointmentForm");
if (form) {
  form.addEventListener("submit", (e) => {
    e.preventDefault();

    const name = form.name.value.trim();
    const phone = form.phone.value.trim();
    const age = form.age.value.trim();
    const service = form.service.value.trim();
    const date = form.date.value;
    const time = form.time.value;
    const message = form.message.value.trim();

    if (!name || !phone || !age || !service || !date || !time) {
      alert("Please fill all required fields.");
      return;
    }

    const waMsg =
      `*Online Appointment Request*\n\n` +
      `*Doctor:* Dr Shahid Iqbal Malik\n` +
      `*Clinic:* Lahore Care Hospital (Main Ferozepur Road, Ghazi Road, Lahore)\n\n` +
      `*Patient Name:* ${name}\n` +
      `*Contact:* ${phone}\n` +
      `*Age:* ${age}\n` +
      `*Concern/Service:* ${service}\n` +
      `*Preferred Date:* ${date}\n` +
      `*Preferred Time:* ${time}\n` +
      (message ? `\n*Message:* ${message}\n` : `\n`) +
      `\nPlease confirm appointment. JazakAllah.`;

    openWhatsApp(waMsg);
    form.reset();
  });
}

// Mobile menu toggle
const hamburger = document.getElementById("hamburger");
const mobileNav = document.getElementById("mobileNav");

if (hamburger && mobileNav) {
  hamburger.addEventListener("click", () => {
    const isHidden = mobileNav.hasAttribute("hidden");
    if (isHidden) {
      mobileNav.removeAttribute("hidden");
      hamburger.setAttribute("aria-expanded", "true");
      hamburger.textContent = "✕";
    } else {
      mobileNav.setAttribute("hidden", "");
      hamburger.setAttribute("aria-expanded", "false");
      hamburger.textContent = "☰";
    }
  });

  mobileNav.querySelectorAll("a").forEach(a => {
    a.addEventListener("click", () => {
      mobileNav.setAttribute("hidden", "");
      hamburger.setAttribute("aria-expanded", "false");
      hamburger.textContent = "☰";
    });
  });
}

// Mail status (after mail.php redirect)
const mailStatus = document.getElementById("mailStatus");
if (mailStatus) {
  const params = new URLSearchParams(window.location.search);
  const s = params.get("mail");
  if (s === "success") {
    mailStatus.textContent = "✅ Message sent successfully.";
  } else if (s === "fail") {
    mailStatus.textContent = "❌ Message not sent. Please try again or WhatsApp.";
  } else {
    mailStatus.textContent = "";
  }
}
